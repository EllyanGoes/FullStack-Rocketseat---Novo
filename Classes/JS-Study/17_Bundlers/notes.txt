#5 como incluir o css no empacotamento 
no webpack 
instalar o styleloader e o css loader e configurar de acordo com o site a baixo 
https://webpack.js.org/loaders/style-loader/#root
module: {
    rules: [
      {
        test: /\.css$/i,
        use: ["style-loader", "css-loader"],
        exclude: "/node_modules",
      },
    ],
  },
